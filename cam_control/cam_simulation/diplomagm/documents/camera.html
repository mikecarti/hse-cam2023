163                 id,
164                 focal_length,
165                 f_number,
166                 focus_distance,
167                 diameter_of_lens,
168                 lenght_of_lens,
169                 width_of_image_sensor,
170                 height_of_image_sensor,
171                 pixel_size,
172                 body_lenght,
173                 body_width,
174                 body_height,
175                 x=0.0,
176                 y=0.0,
177                 z=0.0,
178                 pitch=0.0,
179                 yaw=0.0,
180                 roll=0.0):
181        """_summary_
182
183        :param id: _description_
184        :type id: _type_
185        :param focal_length: _description_
186        :type focal_length: _type_
187        :param f_number: _description_
188        :type f_number: _type_
189        :param focus_distance: _description_
190        :type focus_distance: _type_
191        :param diameter_of_lens: _description_
192        :type diameter_of_lens: _type_
193        :param lenght_of_lens: _description_
194        :type lenght_of_lens: _type_
195        :param width_of_image_sensor: _description_
196        :type width_of_image_sensor: _type_
197        :param height_of_image_sensor: _description_
198        :type height_of_image_sensor: _type_
199        :param pixel_size: _description_
200        :type pixel_size: _type_
201        :param body_lenght: _description_
202        :type body_lenght: _type_
203        :param body_width: _description_
204        :type body_width: _type_
205        :param body_height: _description_
206        :type body_height: _type_
207        :param x: _description_, defaults to 0.0
208        :type x: float, optional
209        :param y: _description_, defaults to 0.0
210        :type y: float, optional
211        :param z: _description_, defaults to 0.0
212        :type z: float, optional
213        :param pitch: _description_, defaults to 0.0
214        :type pitch: float, optional
215        :param yaw: _description_, defaults to 0.0
216        :type yaw: float, optional
217        :param roll: _description_, defaults to 0.0
218        :type roll: float, optional
219        """
220        self.__id = id
221        self.__size = Size(body_lenght, body_width, body_height)
222        self.__coordinates = Coordinates(x, y, z)
223        self.__axis = Axis(pitch, yaw, roll)
224        self.__rotationMatrix = getRotationMatrix(pitch, yaw, roll)
225        self.__vector_x = np.dot(np.array([1, 0, 0]), self.__rotationMatrix)
226        self.__vector_y = np.dot(np.array([0, 1, 0]), self.__rotationMatrix)
227        self.__vector_z = np.dot(np.array([0, 0, 1]), self.__rotationMatrix)
228        self.__lens = Lens(focal_length, f_number, focus_distance,
229                           diameter_of_lens, lenght_of_lens)
230        self.__imageSensor = ImageSensor(width_of_image_sensor,
231                                         height_of_image_sensor, pixel_size)
232        self.__width_angle_of_view = 2 * math.atan(
233            width_of_image_sensor / 2 / focal_length) * 180 / math.pi
234        self.__height_angle_of_view = 2 * math.atan(
235            height_of_image_sensor / 2 / focal_length) * 180 / math.pi
236        self.__hyperfocal_distance = focal_length**2 / f_number / (
237            math.sqrt(2) * pixel_size / 1000)
