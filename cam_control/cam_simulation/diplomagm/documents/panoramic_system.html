 35def toMeters(value):
 36    """Conversion function from millimeters to meters.
 37
 38    :param value: some value in millimeters
 39    :type value: float
 40    :return: the same value in meters
 41    :rtype: float
 42    """

126
127        :param id: id number of panoramic system in list of panoramic systems
128        :type id: int
129        :param body_lenght: length of panoramic systems body, defaults to 0.0
130        :type body_lenght: float, optional
131        :param body_width: width of panoramic systems body, defaults to 0.0
132        :type body_width: float, optional
133        :param body_height: height of panoramic systems body, defaults to 0.0

158        self.__list_of_cameras_rotation_matrices = []

414    def calculatePanoramicSystemFOV(self):
415        """Calculate FOV for each camera in panoramic system and return list of cameras FOV for panoramic system. 
416
417        :return: list of cameras FOV for panoramic system
418        :rtype: list
419        """
420
421        list_of_cameras_FOV = []
422
423        earth_circumference = 40000000
424        self.__list_of_cameras_fov.clear()
425        self.__list_of_cameras_axis.clear()
426
427        A = 0
428        B = 0
429        C = 1
430        D = 0
431
432        vector_plane = np.array([A, B, C])
433
434        panoramic_system_x, panoramic_system_y, panoramic_system_z = self.getCoordinates(
435        )
436        panoramic_system_coordinates = np.array(
437            [panoramic_system_x, panoramic_system_y, panoramic_system_z])
438        panoramic_system_pitch, panoramic_system_yaw, panoramic_system_roll = self.getAxis(
439        )
440        panoramic_system_rotation_matrix = self.getRotationMatrix()
441
442        for camera in self.getListOfCameras():
443            camera_z = toMeters(camera.getCoordinatesZ())
444            camera_x = toMeters(camera.getCoordinatesX())
445            camera_y = toMeters(camera.getCoordinatesY())
446            camera_coordinates = np.array([camera_x, camera_y, camera_z])
447            camera_rotation_matrix = camera.getRotationMatrix()
448            camera_height_angle_of_view = camera.getHeightAngleOfView()
449            camera_width_angle_of_view = camera.getWidthAngleOfView()
450
451            vector_a = np.array([
452                1.0,
453                math.tan(math.radians(camera_width_angle_of_view / 2)),
454                -math.tan(math.radians(camera_height_angle_of_view / 2))
455            ])
456            vector_b = np.array([
457                1.0,
458                math.tan(math.radians(camera_width_angle_of_view / 2)),
459                math.tan(math.radians(camera_height_angle_of_view / 2))
460            ])
461            vector_c = np.array([
462                1.0, -math.tan(math.radians(camera_width_angle_of_view / 2)),
463                math.tan(math.radians(camera_height_angle_of_view / 2))
464            ])
465            vector_d = np.array([
466                1.0, -math.tan(math.radians(camera_width_angle_of_view / 2)),
467                -math.tan(math.radians(camera_height_angle_of_view / 2))
468            ])
469
470            vector_p = np.array([1, 0, 0])
471
472            vector_a = np.dot(
473                np.dot(panoramic_system_rotation_matrix,
474                       camera_rotation_matrix), vector_a)
475            vector_b = np.dot(
476                np.dot(panoramic_system_rotation_matrix,
477                       camera_rotation_matrix), vector_b)
478            vector_c = np.dot(
479                np.dot(panoramic_system_rotation_matrix,
480                       camera_rotation_matrix), vector_c)
481            vector_d = np.dot(
482                np.dot(panoramic_system_rotation_matrix,
483                       camera_rotation_matrix), vector_d)
484            vector_p = np.dot(
485                np.dot(panoramic_system_rotation_matrix,
486                       camera_rotation_matrix), vector_p)
487
488            vector_t_0 = panoramic_system_coordinates + np.dot(
489                panoramic_system_rotation_matrix, camera_coordinates)
490
491            a_0 = -(np.dot(vector_plane, vector_t_0) + D) / (np.dot(
492                vector_plane, vector_a))
493            b_0 = -(np.dot(vector_plane, vector_t_0) + D) / (np.dot(
494                vector_plane, vector_b))
495            c_0 = -(np.dot(vector_plane, vector_t_0) + D) / (np.dot(
496                vector_plane, vector_c))
497            d_0 = -(np.dot(vector_plane, vector_t_0) + D) / (np.dot(
498                vector_plane, vector_d))
499            p_0 = -(np.dot(vector_plane, vector_t_0) + D) / (np.dot(
500                vector_plane, vector_p))
501
502            if (a_0 < 0):
503                a_0 = earth_circumference
504
505            if (b_0 < 0):
506                b_0 = earth_circumference
507
508            if (c_0 < 0):
509                c_0 = earth_circumference
510
511            if (d_0 < 0):
512                d_0 = earth_circumference
513
514            if (p_0 < 0):
515                p_0 = earth_circumference
516
517            fov_a = np.dot(a_0, vector_a) + vector_t_0
518            fov_b = np.dot(b_0, vector_b) + vector_t_0
519            fov_c = np.dot(c_0, vector_c) + vector_t_0
520            fov_d = np.dot(d_0, vector_d) + vector_t_0
521            main_axis = np.dot(p_0, vector_p) + vector_t_0
522
523            camera_fov = Tetragon(fov_a[0], fov_a[1], fov_a[2], fov_b[0],
524                                  fov_b[1], fov_b[2], fov_c[0], fov_c[1],
525                                  fov_c[2], fov_d[0], fov_d[1], fov_d[2])
526
527            self.__list_of_cameras_fov.append(camera_fov)
528
529            camera_main_axis = np.array([
530                Coordinates(vector_t_0[0], vector_t_0[1], vector_t_0[2]),
531                Coordinates(main_axis[0], main_axis[1], main_axis[2])
532            ])
533            self.__list_of_cameras_axis.append(camera_main_axis)
534
535            list_of_cameras_FOV.append([fov_a, fov_b, fov_c, fov_d])
536
537        return list_of_cameras_FOV

593                math.tan(math.radians(camera_height_angle_of_view / 2))

595            vector_d = np.array([
596                1.0, -math.tan(math.radians(camera_width_angle_of_view / 2)),
597                -math.tan(math.radians(camera_height_angle_of_view / 2))
