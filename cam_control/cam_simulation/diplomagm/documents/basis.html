158class Tetragon():
159    """Class for cameras fields of view and sharpness.

162    def __init__(self,
163                 a_x=0.0,
164                 a_y=0.0,
165                 a_z=0.0,
166                 b_x=0.0,
167                 b_y=0.0,
168                 b_z=0.0,
169                 c_x=0.0,
170                 c_y=0.0,
171                 c_z=0.0,
172                 d_x=0.0,
173                 d_y=0.0,
174                 d_z=0.0):
175        """Class constructor.
176
177        :param a_x: x coordinate for vertex a of a quadrilateral, defaults to 0.0
178        :type a_x: float, optional
179        :param a_y: y coordinate for vertex a of a quadrilateral, defaults to 0.0
180        :type a_y: float, optional
181        :param a_z: z coordinate for vertex a of a quadrilateral, defaults to 0.0
182        :type a_z: float, optional
183        :param b_x: x coordinate for vertex b of a quadrilateral, defaults to 0.0
184        :type b_x: float, optional
185        :param b_y: y coordinate for vertex b of a quadrilateral, defaults to 0.0
186        :type b_y: float, optional
187        :param b_z: z coordinate for vertex b of a quadrilateral, defaults to 0.0
188        :type b_z: float, optional
189        :param c_x: x coordinate for vertex c of a quadrilateral, defaults to 0.0
190        :type c_x: float, optional
191        :param c_y: y coordinate for vertex c of a quadrilateral, defaults to 0.0
192        :type c_y: float, optional
193        :param c_z: z coordinate for vertex c of a quadrilateral, defaults to 0.0
194        :type c_z: float, optional
195        :param d_x: x coordinate for vertex d of a quadrilateral, defaults to 0.0
196        :type d_x: float, optional
197        :param d_y: y coordinate for vertex d of a quadrilateral, defaults to 0.0
198        :type d_y: float, optional
199        :param d_z: z coordinate for vertex d of a quadrilateral, defaults to 0.0
200        :type d_z: float, optional
201        """
202        self.__a = Coordinates(a_x, a_y, a_z)
203        self.__b = Coordinates(b_x, b_y, b_z)
204        self.__c = Coordinates(c_x, c_y, c_z)
205        self.__d = Coordinates(d_x, d_y, d_z)
206
207        a = np.array([a_x, a_y, a_z])
208        b = np.array([b_x, b_y, b_z])
209        c = np.array([c_x, c_y, c_z])
210        d = np.array([d_x, d_y, d_z])
211
212        self.__ab = Line(a, b)
213        self.__bc = Line(b, c)
214        self.__cd = Line(c, d)
215        self.__da = Line(d, a)
216
217    def getLines(self):
218        """Get ab, bc, cd, da lines.
219
220        :return: ab, bc, cd, da lines.
221        :rtype: class Line
222        """
223        return self.__ab.getVector(), self.__ab.getPoint(), self.__bc.getVector(
224        ), self.__bc.getPoint(), self.__cd.getVector(), self.__cd.getPoint(
225        ), self.__da.getVector(), self.__da.getPoint()
226

233        return self.__a, self.__b, self.__c, self.__d

275                    c_y=0.0,
276                    c_z=0.0,

278                    d_y=0.0,
279                    d_z=0.0):
280        """Set new coordinates of a, b, c, d vertices.

282        :param a_x: x coordinate for vertex a of a quadrilateral, defaults to 0.0
283        :type a_x: float, optional
284        :param a_y: y coordinate for vertex a of a quadrilateral, defaults to 0.0
285        :type a_y: float, optional
286        :param a_z: z coordinate for vertex a of a quadrilateral, defaults to 0.0
287        :type a_z: float, optional
288        :param b_x: x coordinate for vertex b of a quadrilateral, defaults to 0.0
289        :type b_x: float, optional
290        :param b_y: y coordinate for vertex b of a quadrilateral, defaults to 0.0
291        :type b_y: float, optional
292        :param b_z: z coordinate for vertex b of a quadrilateral, defaults to 0.0
293        :type b_z: float, optional
